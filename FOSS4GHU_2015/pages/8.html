clippedLayer.on('precompose', function (evt) {
                            var ctx = evt.context;
                            ctx.save();
                            ctx.beginPath();
                            ctx.rect(20, 20, 100, 100);
                            ctx.clip();
                        });
                        clippedLayer.on('postcompose', function (evt) {
                            evt.context.restore();
                        });
